00415 template <class TypeReal>
00416 void Matrix3<TypeReal>::orthogonalInPlace() //primaryAxis, secondaryAxis)
00417 {
00418     // Code take it from WildMagig 5  -  www.geometrictools.com  -  here the matrix is transpose
00419     // Algorithm uses Gram-Schmidt orthogonalization.  If 'this' matrix is
00420     // M = [m0|m1|m2], then orthonormal output matrix is Q = [q0|q1|q2],
00421     //
00422     //   q0 = m0/|m0|
00423     //   q1 = (m1-(q0*m1)q0)/|m1-(q0*m1)q0|
00424     //   q2 = (m2-(q0*m2)q0-(q1*m2)q1)/|m2-(q0*m2)q0-(q1*m2)q1|
00425     //
00426     // where |V| indicates length of vector V and A*B indicates dot
00427     // product of vectors A and B.
00428 
00429 
00430 
00431     // Compute q0. length xAxis
00432     TypeReal invLength = (TypeReal)1 / sqrt(data[0]*data[0] +
00433         data[1]*data[1] + data[2]*data[2]);
00434 
00435     data[0] *= invLength;
00436     data[1] *= invLength;
00437     data[2] *= invLength;
00438 
00439     // Compute q1.
00440     TypeReal dot0 = data[0]*data[3] + data[1]*data[4] +
00441         data[2]*data[5];
00442 
00443     data[3] -= dot0*data[0];
00444     data[4] -= dot0*data[1];
00445     data[5] -= dot0*data[2];
00446 
00447     invLength = (TypeReal)1 / sqrt(data[3]*data[3] +
00448         data[4]*data[4] + data[5]*data[5]);
00449 
00450     data[3] *= invLength;
00451     data[4] *= invLength;
00452     data[5] *= invLength;
00453 
00454     // compute q2
00455     TypeReal dot1 = data[3]*data[6] + data[4]*data[7] +
00456         data[5]*data[8];
00457 
00458     dot0 = data[0]*data[6] + data[1]*data[7] +
00459         data[2]*data[8];
00460 
00461     data[6] -= dot0*data[0] + dot1*data[3];
00462     data[7] -= dot0*data[1] + dot1*data[4];
00463     data[8] -= dot0*data[2] + dot1*data[5];
00464 
00465     invLength = (TypeReal)1 / sqrt(data[6]*data[6] +
00466         data[7]*data[7] + data[8]*data[8]);
00467 
00468     data[6] *= invLength;
00469     data[7] *= invLength;
00470     data[8] *= invLength;
00471 }

00802 void Matrix3<TypeReal>::setFromVectorToVector(const Vector3<TypeReal> &fromVec, const Vector3<TypeReal> &toVec)
00803 {

00805     TypeReal e = fromVec.dot(toVec);

00807 
00808     if (f > 1.0-Math<TypeReal>::EPSILON) // "from" and "to" vectors parallel or almost parallel
00809     {

00811         TypeReal fy = fabs(fromVec.y);
00812         TypeReal fz = fabs(fromVec.z);
00813 
00814         if (fx<fy)
00815         {
00816             if (fx<fz) {
00817                 x.set((TypeReal)1.0, (TypeReal)0.0, (TypeReal)0.0);
00818             }
00819             else {
00820                 x.set((TypeReal)0.0, (TypeReal)0.0, (TypeReal)1.0);
00821             }
00822         }
00823         else
00824         {
00825             if (fy<fz) {
00826                 x.set((TypeReal)0.0, (TypeReal)1.0, (TypeReal)0.0);
00827             }
00828             else {
00829                 x.set((TypeReal)0.0, (TypeReal)0.0, (TypeReal)1.0);
00830             }
00831         }
00832 
00833         u = x - fromVec;
00834         v = x - toVec;
00835 
00836         TypeReal c1 = (TypeReal)2.0/(u.dot(u));
00837         TypeReal c2 = (TypeReal)2.0/(v.dot(v));
00838         TypeReal c3 = v.dot(u*(c1*c2));
00839 
00840         TypeReal uvals[3];
00841         TypeReal vvals[3];
00842         uvals[0]=u.x; uvals[1]=u.y; uvals[2]=u.z;
00843         vvals[0]=v.x; vvals[1]=v.y; vvals[2]=v.z;
00844         for (unsigned int i=0; i<3; i++)
00845         {
00846             for (unsigned int j=0; j<3; j++)
00847             {
00848                 this->data[i*3+j] =  - c1*uvals[i]*uvals[j] - c2*vvals[i]*vvals[j] + c3*vvals[i]*uvals[j];
00849             }
00850             this->data[i*4] += (TypeReal)1.0;
00851         }
00852     }
00853     else  // the most common case, unless "from"="to", or "from"=-"to"
00854     {
00855         v = fromVec.cross(toVec);
00856         TypeReal h = (TypeReal)1.0/((TypeReal)1.0 + e);    // optimization by Gottfried Chen
00857         TypeReal hvx = h*v.x;
00858         TypeReal hvz = h*v.z;
00859         TypeReal hvxy = hvx*v.y;
00860         TypeReal hvxz = hvx*v.z;
00861         TypeReal hvyz = hvz*v.y;
00862 
00863 
00864         this->data[0] = e + hvx*v.x;
00865         this->data[1] = hvxy - v.z;
00866         this->data[2] = hvxz + v.y;
00867 
00868         this->data[3] = hvxy + v.z;
00869         this->data[4] = e + h*v.y*v.y;
00870         this->data[5] = hvyz - v.x;
00871 
00872         this->data[6] = hvxz - v.y;
00873         this->data[7] = hvyz + v.x;
00874         this->data[8] = e + hvz*v.z;
00875     }
00876 }
