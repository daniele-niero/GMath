298  void Quaternion::fromMatrix3(const Matrix3& mat)
299  {
300  // Code from Geometric Tools www.geometrictools.com
301  // Algorithm in Ken Shoemake's article in 1987 SIGGRAPH course notes
302  // article "HQuaternion Calculus and Fast Animation".
303 
304  const int next[3] = { 1, 2, 0 };
305 
306  double trace = mat(0,0) + mat(1,1) + mat(2,2);
307  double root;
308 
309  if (trace > 0)
310  {
311  // |w| > 1/2, may as well choose w > 1/2
312  root = sqrt(trace + 1.0); // 2w
313  w = 0.5*root;
314  root = 0.5/root; // 1/(4w)
315  x = (mat(1,2) - mat(2,1))*root;
316  y = (mat(2,0) - mat(0,2))*root;
317  z = (mat(0,1) - mat(1,0))*root;
318  }
319  else
320  {
321  // |w| <= 1/2
322  int i = 0;
323  if (mat(1,1) > mat(0,0))
324  {
325  i = 1;
326  }
327  if (mat(2,2) > mat(i,i))
328  {
329  i = 2;
330  }
331  int j = next[i];
332  int k = next[j];
333 
334  root = sqrt(mat(i,i) - mat(j,j) - mat(k,k) + 1.0);
335  double* quat[3] = { &x, &y, &z };
336  *quat[i] = 0.5*root;
337  root = 0.5/root;
338  w = (mat(j,k) - mat(k,j))*root;
339  *quat[j] = (mat(i,j) + mat(j,i))*root;
340  *quat[k] = (mat(i,k) + mat(k,i))*root;
341  }
342  }

597  {
598  // If q = A*(x*i+y*j+z*k) where (x,y,z) is unit length, then
599  // exp(q) = cos(A)+sin(A)*(x*i+y*j+z*k). If sin(A) is near zero,
600  // use exp(q) = cos(A)+A*(x*i+y*j+z*k) since A/sin(A) has limit 1.

610  {
611  double coeff = sn/angle;
612 
613  result.x = coeff*x;
614  result.y = coeff*y;
615  result.z = coeff*z;
616  //result.w = coeff*w;
617  }
618  else
619  {
620  result.x = x;
621  result.y = y;
622  result.z = z;
623  //result.w = w;
